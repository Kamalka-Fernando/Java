    private void jButton1MousePressed(java.awt.event.MouseEvent evt) {                                      
       String nm=name.getText();
       String ad=address.getText();
       String idno=idNo.getText();
       int phoneno=Integer.parseInt(tpNo.getText());
       String scl=school.getText();
        
     
       
        DefaultTableModel model = (DefaultTableModel) customerTable.getModel();
        
        Object[] tableRow={nm,ad,idno,phoneno,scl};
        
        model.addRow(tableRow);
        if(name.equals(name))
            name.setText(null);
         if(address.equals(address))
           address.setText(null);
          if(idNo.equals(idNo))
            idNo.setText(null);
           if(tpNo.equals(tpNo))
            tpNo.setText(null);
            if(school.equals(school))
            school.setText(null);   
            
    }                                     

    private void customerTableMouseClicked(java.awt.event.MouseEvent evt) {                                           
       int numRows = customerTable.getSelectedRows().length;
       for(int i=0;i<numRows;i++){
       model.removeRow(customerTable.getSelectedRow());
       }
       
        
    
    }
